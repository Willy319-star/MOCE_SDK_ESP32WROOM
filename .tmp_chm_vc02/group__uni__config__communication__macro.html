宏定义

#define USE_UNIONE_PROTOCOL   0
 0: don't use, 1: use
 
#define USE_SAMPLE_PROTOCOL   1
 0: don't use, 1: use (cannot set to 0 if USE_UNIONE_PROTOCOL is 1)
 
#define OPEN_ASR_EVENT_UCP   1
 0: not out, 1: output ASR result by UCP or SUCP uart port
 
#define UNI_UART_DEVICE_NUM   1
 Only 1 can be used on HB-M board
 
#define UNI_UART_PIN_SELECT   0
 0: B7/B6, 1: B0/B1, 2: B2/B3
 
#define UNI_UART_BAUD_RATE   115200
 4800/9600/19200/38400/57600/115200
 
#define UNI_UART_PARITY   0
 0: None, 1: Odd, 2: Even
 
#define UNI_UART_STOP_BIT   1
 1/2
 
#define UNI_UART_DATA_BIT   8
